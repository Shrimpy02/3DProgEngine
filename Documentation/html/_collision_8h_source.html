204 // Helper functions to create default cube meshe based on existing mesh vertex geometry.
205 class Mesh* CreateCollisionSphereFromMesh(class Material* _material, std::vector<struct Vertex>& _existingMesh)
206 {
207 // default extent init
208 glm::vec3 maxExtent(0);
209 glm::vec3 minExtent(0);
210 glm::vec3 center(0);
211 float largetsDiff(0.f);
212
213 // For each
214 for (int i = 0; i < 3; i++)
215 {
216 glm::vec3 pos(0);
217
218 // check all vertices looking for the longest vertex away.
219 for (int j = 0; j < _existingMesh.size(); j++)
220 {
221 // Gets max extent for axis
222 if (_existingMesh[j].mPosition[i] > maxExtent[i])
223 maxExtent[i] = _existingMesh[j].mPosition[i];
224
225 // Gets min extent for axis
226 else if (_existingMesh[j].mPosition[i] < minExtent[i])
227 minExtent[i] = _existingMesh[j].mPosition[i];
228
229 // Gets center location for axis
230 pos[i] = _existingMesh[j].mPosition[i];
231
232 // Finds the vertex that is the furthest from the center
233 if (largetsDiff < abs(_existingMesh[j].mPosition[i]))
234 largetsDiff = abs(_existingMesh[j].mPosition[i]);
235 }
236 center += pos;
237 }
238
239 // Divide by num vertices for average location of center
240 center /= static_cast<float>(_existingMesh.size());
241
242 // set the class values to the calculated values
243 mRadius = (float)largetsDiff * 1.5;
244
245 std::vector<Vertex> vertices;
246 std::vector<Index> indices;
247 Mesh::GenSphere(vertices,indices,2,mRadius);
248
249 return new Mesh("CollisionSphere", std::move(vertices), std::move(indices), _material);
250 }
